# Logical-operator.java

/*Logical Operators */

import java.util.*;

class Logical{

public static void main(String[] args){
  
boolean a = true; 

boolean b= false; 

System.out.println("Logical and :"+(a&&b));

System.out.println("Logical or :"+(a||b));

System.out.println("Logical not  :"+(!a));

}

}
